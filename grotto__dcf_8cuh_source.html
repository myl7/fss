95 void Preprocess(ParityTree &pt, int4 s0, const Cw cws[]) {
96 constexpr size_t N = 1ULL << in_bits;
97
98 // Phase 1: expand tree, write leaf control bits to pt.p[N-1 .. 2N-2]
99 ExpandTree(pt.b, s0, cws, pt.p + (N - 1));
100
101 // Phase 2a: build parity segment tree bottom-up
102 for (size_t j = N - 2; j < N - 1; --j) {
103 pt.p[j] = pt.p[2 * j + 1] ^ pt.p[2 * j + 2];
104 }
105 }

117 __host__ __device__ static bool Eval(const ParityTree &pt, In x) {
118 constexpr size_t N = 1ULL << in_bits;
119 In e = static_cast<In>(x) + 1;
120
121 // e == 0 means x + 1 overflowed, i.e., e = N (entire domain)
122 if (e == 0 || e == N) return pt.p[0];
123
124 bool pi = false;
125 size_t cur = 0;
126 for (int i = 0; i < in_bits; ++i) {
127 bool e_bit = (e >> (in_bits - 1 - i)) & 1;
128 if (e_bit) {
129 pi ^= pt.p[2 * cur + 1];
130 cur = 2 * cur + 2;
131 } else {
132 cur = 2 * cur + 1;
133 }
134 }
135 return pi;
136 }

152 void EvalAll(bool b, int4 s0, const Cw cws[], bool ys[]) {
153 constexpr size_t N = 1ULL << in_bits;
154
155 // Phase 1: expand tree to get leaf control bits into ys[]
156 ExpandTree(b, s0, cws, ys);
157
158 // Phase 2b: prefix-sum scan (running XOR)
159 // ys[x] currently holds leaf x's control bit.
160 // Transform to: ys[x] = XOR of control bits [0..x] = share of 1[alpha <= x].
161 for (size_t x = 1; x < N; ++x) {
162 ys[x] = ys[x] ^ ys[x - 1];
163 }
164 }

175 void ExpandTree(bool b, int4 s0, const Cw cws[], bool t[]) {
176 int4 st = s0;
177 st = util::SetLsb(st, b);
178
179 assert(in_bits < sizeof(size_t) * 8);
180 size_t l = 0;
181 size_t r = 1ULL << in_bits;
182 int i = 0;
183
184 int par_depth_ = 0;
185 if constexpr (par_depth == -1) {
186 int threads = omp_get_max_threads();
187 while ((1 << par_depth_) < threads) {
188 par_depth_++;
189 }
190 } else {
191 par_depth_ = par_depth;
192 }
193
194#pragma omp parallel
195#pragma omp single
196 ExpandTreeRec(st, cws, t, l, r, i, par_depth_);

199 void ExpandTreeRec(
200 int4 st, const Cw cws[], bool t[], size_t l, size_t r, int i, int par_depth_) {
201 bool tc = util::GetLsb(st);
202 int4 s = st;
203 s = util::SetLsb(s, false);
204
205 if (i == in_bits) {
206 assert(l + 1 == r);
207 t[l] = tc;
208 return;
209 }
210
211 Cw cw = cws[i];
212 int4 s_cw = cw.s;
213 bool tl_cw = util::GetLsb(s_cw);
214 s_cw = util::SetLsb(s_cw, false);
215 bool tr_cw = cw.tr;
216
217 auto [sl, sr] = prg.Gen(s);
218
219 bool tl = util::GetLsb(sl);
220 sl = util::SetLsb(sl, false);
221 bool tr = util::GetLsb(sr);
222 sr = util::SetLsb(sr, false);
223
224 if (tc) {
225 sl = util::Xor(sl, s_cw);
226 sr = util::Xor(sr, s_cw);
227 tl = tl ^ tl_cw;
228 tr = tr ^ tr_cw;
229 }
230
231 int4 stl = sl;
232 stl = util::SetLsb(stl, tl);
233 int4 str = sr;
234 str = util::SetLsb(str, tr);
235
236 size_t mid = (l + r) / 2;
237
238 if (i < par_depth_) {
239#pragma omp task
240 ExpandTreeRec(stl, cws, t, l, mid, i + 1, par_depth_);
241#pragma omp task
242 ExpandTreeRec(str, cws, t, mid, r, i + 1, par_depth_);
243#pragma omp taskwait
244 } else {
245 ExpandTreeRec(stl, cws, t, l, mid, i + 1, par_depth_);
246 ExpandTreeRec(str, cws, t, mid, r, i + 1, par_depth_);
247 }
248 }
