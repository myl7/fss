48static __m128i aes_128_key_expansion(__m128i key, __m128i keygened) {
49 keygened = _mm_shuffle_epi32(keygened, _MM_SHUFFLE(3, 3, 3, 3));
50 key = _mm_xor_si128(key, _mm_slli_si128(key, 4));
51 key = _mm_xor_si128(key, _mm_slli_si128(key, 4));
52 key = _mm_xor_si128(key, _mm_slli_si128(key, 4));
53 return _mm_xor_si128(key, keygened);
54}

71static void aes128_load_key(const uint8_t *enc_key, __m128i *key_schedule) {
72 aes128_load_key_enc_only(enc_key, key_schedule);
73
74 // generate decryption keys in reverse order.
75 // k[10] is shared by last encryption and first decryption rounds
76 // k[0] is shared by first encryption round and last decryption round (and is the original user key)
77 // For some implementation reasons, decryption key schedule is NOT the encryption key schedule in reverse order
78 key_schedule[11] = _mm_aesimc_si128(key_schedule[9]);
79 key_schedule[12] = _mm_aesimc_si128(key_schedule[8]);
80 key_schedule[13] = _mm_aesimc_si128(key_schedule[7]);
81 key_schedule[14] = _mm_aesimc_si128(key_schedule[6]);
82 key_schedule[15] = _mm_aesimc_si128(key_schedule[5]);
83 key_schedule[16] = _mm_aesimc_si128(key_schedule[4]);
84 key_schedule[17] = _mm_aesimc_si128(key_schedule[3]);
85 key_schedule[18] = _mm_aesimc_si128(key_schedule[2]);
86 key_schedule[19] = _mm_aesimc_si128(key_schedule[1]);
87}

89static void aes128_enc(__m128i *key_schedule, const uint8_t *plainText, uint8_t *cipherText) {
90 __m128i m = _mm_loadu_si128((const __m128i *)plainText);
91
92 DO_ENC_BLOCK(m, key_schedule);
93
94 _mm_storeu_si128((__m128i *)cipherText, m);
95}

97static void aes128_dec(__m128i *key_schedule, uint8_t *cipherText, uint8_t *plainText) {
98 __m128i m = _mm_loadu_si128((__m128i *)cipherText);
99
100 DO_DEC_BLOCK(m, key_schedule);
101
102 _mm_storeu_si128((__m128i *)plainText, m);
103}

109static int aes128_self_test(void) {
110 uint8_t plain[] = {0x32, 0x43, 0xf6, 0xa8, 0x88, 0x5a, 0x30, 0x8d, 0x31, 0x31, 0x98, 0xa2, 0xe0, 0x37, 0x07, 0x34};
111 uint8_t enc_key[] = {0x2b, 0x7e, 0x15, 0x16, 0x28, 0xae, 0xd2, 0xa6, 0xab, 0xf7, 0x15, 0x88, 0x09, 0xcf, 0x4f, 0x3c};
112 uint8_t cipher[] = {0x39, 0x25, 0x84, 0x1d, 0x02, 0xdc, 0x09, 0xfb, 0xdc, 0x11, 0x85, 0x97, 0x19, 0x6a, 0x0b, 0x32};
113 uint8_t computed_cipher[16];
114 uint8_t computed_plain[16];
115 int out = 0;
116 __m128i key_schedule[20];
117 aes128_load_key(enc_key, key_schedule);
118 aes128_enc(key_schedule, plain, computed_cipher);
119 aes128_dec(key_schedule, cipher, computed_plain);
120 if (memcmp(cipher, computed_cipher, sizeof(cipher))) out = 1;
121 if (memcmp(plain, computed_plain, sizeof(plain))) out |= 2;
122 return out;
123}
