53
54 for (int i = 0; i < mul; ++i) {
55 ctxs[i] = EVP_CIPHER_CTX_new();
56 assert(ctxs[i] != NULL);

58 ret = EVP_EncryptInit_ex2(ctxs[i], EVP_aes_128_ecb(), keys[i], NULL, NULL);
59 assert(ret == 1);

61 ret = EVP_CIPHER_CTX_set_padding(ctxs[i], 0);
62 assert(ret == 1);
63 }

65 }
