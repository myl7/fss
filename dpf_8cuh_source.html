119 bool tr_cw = t0r ^ t1r ^ a_bit;

121 if (!a_bit) {

123 if (t0) s0 = util::Xor(s0, s_cw);
124 s1 = s1l;
125 if (t1) s1 = util::Xor(s1, s_cw);
126

128 else t0 = t0l;
129 if (t1) t1 = t1l ^ tl_cw;
130 else t1 = t1l;
131 } else {
132 s0 = s0r;

134 s1 = s1r;
135 if (t1) s1 = util::Xor(s1, s_cw);
136

138 else t0 = t0r;
139 if (t1) t1 = t1r ^ tr_cw;
140 else t1 = t1r;
141 }
142
143 // s_cw is updated here
144 s_cw = util::SetLsb(s_cw, tl_cw);
145 cws[i] = {s_cw, tr_cw};
146 }
147
148 auto v_cw_np1 = Group::From(b_buf) + (-Group::From(s0)) + Group::From(s1);
149 if (t1) v_cw_np1 = -v_cw_np1;
150 cws[in_bits] = {v_cw_np1.Into(), false};
151 }

162 __host__ __device__ int4 Eval(bool b, int4 s0, const Cw cws[], In x) {
163 int4 s = s0;
164 s = util::SetLsb(s, false);
165 bool t = b;
166
167 for (int i = 0; i < in_bits; ++i) {
168 Cw cw = cws[i];
169 int4 s_cw = cw.s;
170 bool tl_cw = util::GetLsb(s_cw);
171 s_cw = util::SetLsb(s_cw, false);
172 bool tr_cw = cw.tr;
173
174 auto [sl, sr] = prg.Gen(s);
175
176 bool tl = util::GetLsb(sl);
177 sl = util::SetLsb(sl, false);
178 bool tr = util::GetLsb(sr);
179 sr = util::SetLsb(sr, false);
180
181 if (t) {
182 sl = util::Xor(sl, s_cw);
183 sr = util::Xor(sr, s_cw);
184 tl = tl ^ tl_cw;
185 tr = tr ^ tr_cw;
186 }
187
188 bool x_bit = (x >> (in_bits - 1 - i)) & 1;
189
190 if (!x_bit) {
191 s = sl;
192 t = tl;
193 } else {
194 s = sr;
195 t = tr;
196 }
197 }
198
199 auto y = Group::From(s);
200 int4 v_cw_np1 = cws[in_bits].s;
201 assert((v_cw_np1.w & 1) == 0);
202 if (t) y = y + Group::From(v_cw_np1);
203 if (b) y = -y;
204
205 return y.Into();
206 }
