32constexpr double Log2(double x) {
33 if (x <= 0) return -1e308;
34 int e = 0;
35 double m = x;
36 while (m >= 2.0) {
37 m /= 2.0;
38 ++e;
39 }
40 while (m < 1.0) {
41 m *= 2.0;
42 --e;
43 }
44 double y = (m - 1.0) / (m + 1.0);
45 double y2 = y * y, sum = 0, term = y;
46 for (int k = 0; k < 40; ++k) {
47 sum += term / (2 * k + 1);
48 term *= y2;
49 }
50 return e + 2.0 * sum / 0.6931471805599453;
51}

75constexpr int ChBucket(int t, int lambda) {
76 // Remark 1 applies for t >= 30.
77 // For smaller t, the full Lemma 5 formula with Normal CDF factors is needed,
78 // but the paper considers t >= 30 to capture nearly all practical use cases.
79 assert(t >= 30);
80 double td = static_cast<double>(t);
81 double e = (static_cast<double>(lambda) + 130.0 + detail::Log2(td)) / 123.5;
82 return static_cast<int>(detail::Ceil(e * td));
83}

113 std::pair<int, int> Locate(int4 sigma, In x, int k, __uint128_t n, int b_size) {
114 __uint128_t val = static_cast<__uint128_t>(x) + n * k;
115 __uint128_t domain = n * 3; // kappa = 3
116 __uint128_t y = prp.Permu(sigma, val, domain);

134template <typename Prp, typename In, int kappa = 3>
135 requires Permutable<Prp>
136struct Compact {
137 Prp prp;

153 int Run(std::span<const In> as, int m, int4 sigma, __uint128_t n, int b_size, int ch_retry,
154 std::span<std::pair<int, int>> table) {
155 PrpHash<Prp, In> hasher{prp};
156 int t = static_cast<int>(as.size());
157
158 // Initialize table to empty.
159 for (int i = 0; i < m; ++i) {
160 table[i] = {-1, -1};
161 }
162
163 std::mt19937 rng(42);
164
165 for (int omega = 0; omega < t; ++omega) {
166 int cur_idx = omega;
167 int cur_k = static_cast<int>(rng() % kappa);
168 int evictions = 0;
169
170 for (;;) {
171 auto [bucket, _index] = hasher.Locate(sigma, as[cur_idx], cur_k, n, b_size);
172 // Clamp bucket to [0, m).
173 bucket = bucket % m;
174
175 if (table[bucket].first == -1) {
176 // Empty slot found.
177 table[bucket] = {cur_idx, cur_k};
178 break;
179 }
180
181 // Evict the existing entry and take its place.
182 int evicted_idx = table[bucket].first;
183 int evicted_k = table[bucket].second;
184 table[bucket] = {cur_idx, cur_k};
185
186 // The evicted element becomes the current element with a new random hash.
187 cur_idx = evicted_idx;
188 cur_k = static_cast<int>(rng() % kappa);
189
190 ++evictions;
191 if (evictions > ch_retry) {
192 return 1;
193 }
194 }
195 }
196
197 return 0;
198 }
199};
